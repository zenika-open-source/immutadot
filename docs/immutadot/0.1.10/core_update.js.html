  1. import fpUpdate from 'lodash/fp/update'
  2. import isFunction from 'lodash/isFunction'
  3. import { lodashFpConvertOptions } from './consts'
  5. const rawUpdate = fpUpdate.convert(lodashFpConvertOptions)
  7. const updatePassingArgs = (obj, path, fn, ...args) => rawUpdate(obj, path, v => fn(v, ...args))
  9. /**
  10.  * Updates the value at <code>path</code> of <code>object</code> using the <code>updater</code> function.<br/>
  11.  * The updater is invoked with <code>value</code> and <code>…args</code>.<br/>
  12.  * <code>update</code> can be called with <code>updater</code> only, returning a function taking only <code>object</code>, <code>path</code> and <code>…args</code>.
  13.  * @function update
  14.  * @memberof core
  15.  * @param {Object} [object] The object to modify.
  16.  * @param {Array|string} [path] The path of the property to set.
  17.  * @param {function} updater The function to produce the updated value.
  18.  * @param {...*} args The remaining args.
  19.  * @return {Object|function} Returns the updated object or the wrapped function.
  20.  * @example <caption>Updating a prop</caption>
  21.  * const inc = (v, i = 1) => v + i // this function increments a number with an optional value which defaults to 1
  22.  * const object = { nested: { prop: 4 } }
  23.  * update(object, 'nested.prop', inc) // => { nested: { prop: 5 } }
  24.  * update(object, 'nested.prop', inc, 2) // => { nested: { prop: 6 } }
  25.  * @example <caption>Wrapping an updater</caption>
  26.  * const inc = (v, i = 1) => v + i // this function increments a number with an optional value which defaults to 1
  27.  * const incProp = update(inc)
  28.  * const object = { nested: { prop: 4 } }
  29.  * incProp(object, 'nested.prop') // => { nested: { prop: 5 } }
  30.  * incProp(object, 'nested.prop', 2) // => { nested: { prop: 6 } }
  31.  * @see {@link https://lodash.com/docs#update|lodash.update} for more information.
  32.  * @since 0.1.5
  33.  */
  34. export default (...args) => {
  35.   if (args.length === 1 && isFunction(args[0]))
  36.     return (obj, path, ...rest) => updatePassingArgs(obj, path, args[0], ...rest)
  37.   return updatePassingArgs(...args)
  38. }
