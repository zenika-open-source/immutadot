  2. import * as collection from '../collection'

  7. import concat from 'lodash/concat'
  8. import flow from 'lodash/flow'
  9. import mapValues from 'lodash/mapValues'
  10. import toPath from 'lodash/toPath'
  12. /**
  13.  * Wrapper allowing to make sequences of immutadot functions calls on an object.<br/>
  14.  * Instances are created by calling {@link seq.chain}.<br/>
  15.  * The result of such sequences must be unwrapped with {@link seq.ChainWrapper#value}.
  16.  * @memberof seq.
  17.  * @see {@link seq.chain} for more information.
  18.  * @private
  19.  * @since 0.1.8
  20.  */
  21. class ChainWrapper {
  23.   /**
  24.    * This constructor should not be called directly.<br/>
  25.    * Instances are created by calling {@link seq.chain}.
  26.    * @param {Object} object The object to wrap.
  27.    * @param {Array|string} [path] The path of the object on which functions are called.
  28.    * @see {@link seq.chain} for more information.
  29.    * @since 0.1.8
  30.    */
  31.   constructor(object, path) {
  32.     this._object = object
  33.     this._path = path
  34.     this._flow = []
  35.   }
  37.   /**
  38.    * Translates a relative path to an absolute path, using <code>this._path</code> as a base path.
  39.    * @param {Array|string} path A relative path.
  40.    * @return {Array} Absolute path.
  41.    * @since 0.1.11
  42.    */
  43.   _absolutePath(path) {
  44.     return concat(toPath(this._path), toPath(path))
  45.   }
  47.   /**
  48.    * Add a function call to the sequence.
  49.    * @param {function} fn The function to call.
  50.    * @param {Array|string} path The path of the property to be set.
  51.    * @param {...*} args The arguments for the function call.
  52.    * @returns {seq.ChainWrapper} The wrapper instance.
  53.    * @since 0.1.8
  54.    */
  55.   _call(fn, path, args) {
  56.     this._flow.push(object => fn(object, this._absolutePath(path), ...args))
  57.     return this
  58.   }
  60.   /**
  61.    * Executes the chain sequence to resolve the unwrapped object.
  62.    * @returns {Object} Returns the resolved unwrapped object.
  63.    * @example
  64.    * chain({ nested1: { prop: 'old' }, nested2: { prop: 1 } })
  65.    *   .set('nested1.prop', 'new')
  66.    *   .unset('nested2.prop')
  67.    *   .value() // => { nested1: { prop: 'new' }, nested2: {} }
  68.    * @see {@link seq.chain|chain} for more information.
  69.    * @since 0.1.8
  70.    */
  71.   value() {
  72.     return flow(this._flow)(this._object)
  73.   }
  74. }
  76. // Add namespaces functions to the ChainWrapper prototype
  77. [
  78.   array,
  79.   collection,
  80.   core,
  81.   lang,
  82.   math,
  83. ].forEach(namespace => Object.assign(
  84.   ChainWrapper.prototype,
  85.   mapValues(
  86.     namespace,
  87.     fn => function(path, ...args) {
  88.       return this._call(fn, path, args) // eslint-disable-line no-invalid-this
  89.     }
  90.   )
  91. ))
  93. export default ChainWrapper
